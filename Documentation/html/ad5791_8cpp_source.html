33double AD5791::bytesToVoltage(spi_utils::Message message) {
34
35 byte byte1 = message.msg[0];
36 byte byte2 = message.msg[1];
37 byte byte3 = message.msg[2];
38
39 // The conversion below is for two's complement
40 uint32_t decimal = threeByteToInt(byte1, byte2, byte3);
41 double voltage;
42 if (decimal <= 524287) {
43 voltage = decimal * DAC_FULL_SCALE / 524287;
44 }
45 else {
46 voltage = -(1048576 - decimal) * DAC_FULL_SCALE / 524288;
47 }
48 return voltage;

67spi_utils::Message AD5791::setVoltageMsg(double voltage) {
68
69 uint32_t decimal;

76 else {

80 // Check datasheet for details
81 msg.msg[0] = (byte)((decimal >> 16) | 16); // Writes to dac register

83 msg.msg[2] = (byte)(decimal & 255); // Writes second byte
