353spi_utils::Message AD4115::setupConfigMsg(void) {
354
355 spi_utils::Message msg;
356
357 // 100000 -- Address [0:5] (e.g. Setup 0)
358 // 0 -- WRITE [6]
359 // 0 -- WEN [7]
360 msg.msg[0] = 0x20; // Send 0010 0000 == 32
361
362 // Enable/disable input buffers [8:9] -- 11 (e.g. enabled)
363 // Enable/disable REF(-) input buffer [10] -- 0 (e.g. disabled)
364 // Enable/disable REF(+) input buffer [11] -- 0 (e.g. disabled)
365 // Bipolar/unipolar output coding [12] -- 1 (e.g. bipolar)
366 // Reserved [13:15] -- 000
367 msg.msg[1] = 0x13; // Send 0001 0011 = 19
368
369 // Reserved [0:3] -- 0000
370 // Select ref source [4:5] -- 00 (e.g. external ref)
371 // Reserved [6:7] -- 00
372 msg.msg[2] = 0x00; // Send 0000 0000 = 0
373
374 return msg;
